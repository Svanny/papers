\title{Dynamic Latent Order Inference for Session Editing with Cold Item Uncertainty}
\author{Monysovann Ly}
\date{February 21, 2026}

\begin{document}
\maketitle

\begin{abstract}
We study the problem of inferring a user preferred ordering of a finite set of items from interaction traces generated by edit sessions.
The preferred order is treated as a drifting hidden state.
Observations are noisy and partial, and many items are cold in the sense that they receive updates infrequently.
We propose a state space formulation in which latent one dimensional coordinates induce a permutation by sorting, while inactivity increases posterior uncertainty rather than enforcing a deterministic freeze.
We derive the full Bayesian filtering recursion, a Poisson proximity likelihood based on consecutive update counts, and a maximum a posteriori objective that can be solved daily by second order methods.
A Laplace approximation provides calibrated uncertainty estimates, enabling conservative ranking and conservative movement rules that mirror skill rating systems for infrequent competitors.
\end{abstract}

\paragraph{Keywords.}
Dynamic ranking, state space models, Laplace approximation, Poisson likelihood, cold start, uncertainty inflation, conservative ordering.

\section{Definitions and signals}

\subsection{Items and permutations}
Let $[n] := \{1,\dots,n\}$ index items.
An ordering is a permutation $\pi \in \mathfrak S_n$.
Write $\mathrm{pos}_{\pi}(i)\in[n]$ for the position of item $i$ under $\pi$.

\subsection{Sessions and consecutive update events}
A session is a sequence of visited items $(v_1,\dots,v_m)$ with indicators $a_t\in\{0,1\}$ where $a_t=1$ denotes an update event.
Let $t_1<\cdots<t_k$ be indices with $a_{t_r}=1$ and define the update sequence
\[
u_r := v_{t_r},\qquad r=1,\dots,k.
\]
Let $\Delta_r\ge 0$ be the time between update events $r$ and $r+1$.

\subsection{Daily consecutive update adjacency counts}
Fix a robust weight $w(\Delta)\ge 0$.
For day $d$, define symmetric consecutive update counts
\[
\widehat S_{ij}(d)
:=
\sum_{\text{sessions on day }d}\ \sum_{r=1}^{k-1}
\mathbf{1}\{u_r=i,\ u_{r+1}=j\}\,w(\Delta_r)
+
\mathbf{1}\{u_r=j,\ u_{r+1}=i\}\,w(\Delta_r).
\]
This statistic is designed to reflect proximity between items that are edited consecutively, rather than raw swipe adjacency.

\subsection{Cold mass and days since activity}
Let $\widehat c_i(d)$ be the number of update events on day $d$ involving item $i$.
Maintain a decayed mass
\[
c_i(d) = \rho_c c_i(d-1) + (1-\rho_c)\widehat c_i(d),
\qquad 0<\rho_c<1.
\]
Let $t_i(d)\in\{0,1,2,\dots\}$ denote the number of days since item $i$ last had an update event.

\section{Problem construction}

\subsection{Latent preference as a drifting hidden order}
We assume there exists a latent preferred permutation $\pi^\star(d)$ which drifts over days.
We do not model $\pi^\star(d)$ directly as a discrete state.
Instead we represent it through latent coordinates
\[
z(d)\in\mathbb R^n,
\qquad
\pi^\star(d) = \mathrm{argsort}(z(d)).
\]
This continuous relaxation makes filtering tractable while preserving the permutation as the induced ordering.

\subsection{Cold item paradigm}
Cold items do not provide evidence.
The correct response is not to treat them as fixed.
Instead, inactivity should increase uncertainty about their latent coordinates.
This is the same principle used in skill rating systems where infrequent competitors have increased uncertainty between events.

\section{Generative model}

\subsection{Dynamics prior with inactivity driven uncertainty inflation}
We use a Gaussian transition model
\[
z(d)\mid z(d-1)\sim \mathcal N\!\big(z(d-1),\,Q(d)\big),
\qquad
Q(d)=\mathrm{diag}\big(\tau^2 t_1(d),\dots,\tau^2 t_n(d)\big),
\]
where $\tau^2>0$ controls drift scale.
If item $i$ is inactive for many days, $t_i(d)$ is large and the process variance for $z_i(d)$ increases.

\subsection{Poisson proximity likelihood from consecutive update counts}
For $i<j$, we model daily counts as conditionally independent Poisson variables
\[
\widehat S_{ij}(d)\mid z(d)\sim \mathrm{Poisson}\big(\lambda_{ij}(d)\big).
\]
Let $\delta_{ij}(d)=|z_i(d)-z_j(d)|$ be latent distance.
We set an intensity that decays with distance
\[
\lambda_{ij}(d)
=
e_{ij}(d)\,\exp\!\Big(\beta(d)-\alpha\,\delta_{ij}(d)\Big),
\qquad \alpha>0.
\]
Here $\beta(d)$ is a day intercept capturing overall volume, and $e_{ij}(d)$ is an exposure factor.

\subsection{Exposure factors implementing cold item discounting}
To ensure cold items contribute less information rather than being deterministically pinned, define
\[
\mathrm{expo}_i(d) := \frac{c_i(d)}{c_i(d)+\tau_e},
\qquad
e_{ij}(d) := \mathrm{expo}_i(d)\,\mathrm{expo}_j(d),
\qquad \tau_e>0.
\]
If item $i$ is cold, then $\mathrm{expo}_i(d)$ is small and all edges touching $i$ have small exposure.

\subsection{Gauge fixing}
The likelihood depends on differences $z_i-z_j$ and is invariant to translation.
We fix the gauge by enforcing
\[
\mathbf{1}^\top z(d)=0.
\]
Optionally add a weak quadratic anchor to stabilize scale, for example $\|z(d)\|_2^2$ with a small coefficient, or constrain $\|z(d)\|_2=1$.

\section{Bayesian filtering and full posterior update}

Let $y_d:=\widehat S(d)$ denote all daily observations.
The exact Bayesian filter recursion is
\[
p\big(z(d)\mid y_{1:d}\big)
\propto
p\big(y_d\mid z(d)\big)\,p\big(z(d)\mid y_{1:d-1}\big),
\]
with prediction
\[
p\big(z(d)\mid y_{1:d-1}\big)
=
\int p\big(z(d)\mid z(d-1)\big)\,p\big(z(d-1)\mid y_{1:d-1}\big)\,dz(d-1).
\]
These equalities define the full posterior update for a drifting hidden state.

\section{MAP objective and Laplace posterior approximation}

\subsection{Negative log likelihood}
\paragraph{Derivation.}
For a fixed day $d$ and pair $i<j$, assume
\[
\widehat S_{ij}(d)\mid z(d)\sim \mathrm{Poisson}\!\big(\lambda_{ij}(d)\big).
\]
The Poisson pmf is
\[
p\!\left(\widehat S_{ij}(d)\mid z(d)\right)
=
\frac{\lambda_{ij}(d)^{\widehat S_{ij}(d)}\exp\!\big(-\lambda_{ij}(d)\big)}{\widehat S_{ij}(d)!}.
\]
Taking the negative log gives
\[
-\log p\!\left(\widehat S_{ij}(d)\mid z(d)\right)
=
\lambda_{ij}(d)-\widehat S_{ij}(d)\log \lambda_{ij}(d)+\log\!\big(\widehat S_{ij}(d)!\big).
\]
Summing over all pairs $i<j$ yields the daily negative log-likelihood
\[
\mathcal L_d(z)
:=
-\sum_{i<j}\log p\!\left(\widehat S_{ij}(d)\mid z\right)
=
\sum_{i<j}\Big(\lambda_{ij}(d)-\widehat S_{ij}(d)\log \lambda_{ij}(d)\Big)
+\sum_{i<j}\log\!\big(\widehat S_{ij}(d)!\big).
\]
Dropping constants independent of $z$, the Poisson negative log likelihood is
\[
\mathcal L_d(z)
=
\sum_{i<j}\Big(\lambda_{ij}(d) - \widehat S_{ij}(d)\log \lambda_{ij}(d)\Big).
\]

\subsection{Gaussian predicted prior}
Assume the predicted prior is approximated as Gaussian
\[
p\big(z(d)\mid y_{1:d-1}\big)\approx \mathcal N\!\big(m^-(d),P^-(d)\big).
\]
This approximation yields a quadratic prior term in the daily MAP objective.

\paragraph{Derivation (moment matching under a random walk).}
Suppose the previous day's posterior is approximated as
\[
p\big(z(d-1)\mid y_{1:d-1}\big)\approx \mathcal N\!\big(\widehat z(d-1),\Sigma(d-1)\big),
\]
and the transition model is Gaussian,
\[
z(d)\mid z(d-1)\sim \mathcal N\!\big(z(d-1),Q(d)\big).
\]
Then
\[
z(d)=z(d-1)+\varepsilon(d),\qquad \varepsilon(d)\sim \mathcal N\!\big(0,Q(d)\big),
\]
with $\varepsilon(d)$ independent of $z(d-1)$ given $y_{1:d-1}$. Therefore the predicted mean and covariance satisfy
\[
m^-(d)=\mathbb E\!\big[z(d)\mid y_{1:d-1}\big]=\widehat z(d-1),
\qquad
P^-(d)=\mathrm{Cov}\!\big(z(d)\mid y_{1:d-1}\big)=\Sigma(d-1)+Q(d).
\]

\paragraph{Remark (compatibility with the Laplace step).}
Section~5.2 is the \emph{prediction} step: it propagates the previous day's \emph{approximate} posterior
$\mathcal N\!\big(\widehat z(d-1),\Sigma(d-1)\big)$ through the Gaussian dynamics to obtain a Gaussian predicted prior
$\mathcal N\!\big(m^-(d),P^-(d)\big)$ for day $d$.
Section~5.5 is the \emph{update} step: it approximates the (generally non-Gaussian) posterior after observing $y_d$
by a Gaussian centered at the MAP, with covariance given by the inverse Hessian of the day-$d$ negative log posterior.
Thus, there is no requirement that $\Sigma(d)$ equal $P^-(d)$; typically $\Sigma(d)\preceq P^-(d)$ when the day-$d$
likelihood contributes positive curvature.

\subsection{Daily MAP objective}
The MAP estimate $\widehat z(d)$ is obtained by minimizing the negative log posterior
\[
\widehat z(d)
=
\arg\min_{z:\ \mathbf{1}^\top z=0}
\Big[
\mathcal L_d(z)
+\frac12\big(z-m^-(d)\big)^\top \big(P^-(d)\big)^{-1}\big(z-m^-(d)\big)
\Big].
\]
\paragraph{Derivation.}
By Bayes' rule,
\[
p\big(z(d)\mid y_{1:d}\big)\propto p\big(y_d\mid z(d)\big)\,p\big(z(d)\mid y_{1:d-1}\big).
\]
Taking the negative log and dropping additive constants independent of $z$ yields
\[
-\log p\big(z(d)\mid y_{1:d}\big)
\equiv
-\log p\big(y_d\mid z(d)\big)-\log p\big(z(d)\mid y_{1:d-1}\big).
\]
The first term is $\mathcal L_d(z)$ (Section~5.1).
Using the Gaussian approximation from Section~5.2, the second term becomes the quadratic form
\[
\frac12\big(z-m^-(d)\big)^\top\big(P^-(d)\big)^{-1}\big(z-m^-(d)\big)
\]
up to constants.
Finally, the constraint $\mathbf{1}^\top z=0$ fixes the additive shift invariance of $z$.
This objective is the central daily inference problem.

\subsection{Gradient and Hessian structure}
Write $\delta_{ij}=|z_i-z_j|$ and $s_{ij}=\mathrm{sign}(z_i-z_j)$, defined as $s_{ij}\in\{-1,0,1\}$ and treated as constant away from ties.
Then
\[
\frac{\partial \lambda_{ij}}{\partial z_i} = -\alpha s_{ij}\lambda_{ij},
\qquad
\frac{\partial \lambda_{ij}}{\partial z_j} = \alpha s_{ij}\lambda_{ij}.
\]
The gradient of $\mathcal L_d$ is
\[
\frac{\partial \mathcal L_d}{\partial z_i}
=
\sum_{j\ne i}\alpha s_{ij}\big(\widehat S_{ij}(d)-\lambda_{ij}(d)\big).
\]
Away from ties, the Hessian of $\mathcal L_d$ has a graph Laplacian form
\[
\frac{\partial^2 \mathcal L_d}{\partial z_i^2}
=
\sum_{j\ne i}\alpha^2 \lambda_{ij}(d),
\qquad
\frac{\partial^2 \mathcal L_d}{\partial z_i\partial z_j}
=
-\alpha^2 \lambda_{ij}(d)\ \ (i\ne j).
\]
Therefore the posterior curvature contributed by the data is controlled by $\lambda_{ij}(d)$.

\paragraph{Derivation.}
Recall $\lambda_{ij}(d)=e_{ij}(d)\exp\big(\beta(d)-\alpha\delta_{ij}(d)\big)$ with $\delta_{ij}=|z_i-z_j|$.
Away from ties ($z_i\neq z_j$), the derivative of the absolute value is
\[
\frac{\partial \delta_{ij}}{\partial z_i}=\mathrm{sign}(z_i-z_j)=s_{ij},
\qquad
\frac{\partial \delta_{ij}}{\partial z_j}=-s_{ij}.
\]
By the chain rule,
\[
\frac{\partial \lambda_{ij}}{\partial z_i}
=
\lambda_{ij}\,\frac{\partial}{\partial z_i}\big(-\alpha\delta_{ij}\big)
=
-\alpha\lambda_{ij}\,\frac{\partial\delta_{ij}}{\partial z_i}
=
-\alpha s_{ij}\lambda_{ij},
\qquad
\frac{\partial \lambda_{ij}}{\partial z_j}
=
\alpha s_{ij}\lambda_{ij}.
\]
Next, for a single pair $i<j$ define the (non-constant part of the) Poisson negative log likelihood contribution
\[
\ell_{ij}(z):=\lambda_{ij}(d)-\widehat S_{ij}(d)\log\lambda_{ij}(d).
\]
Differentiating and using $\partial_{z_i}\log\lambda_{ij}=(\partial_{z_i}\lambda_{ij})/\lambda_{ij}$ gives
\[
\frac{\partial \ell_{ij}}{\partial z_i}
=
\frac{\partial \lambda_{ij}}{\partial z_i}
-\widehat S_{ij}(d)\,\frac{1}{\lambda_{ij}}\frac{\partial \lambda_{ij}}{\partial z_i}
=
\Big(1-\frac{\widehat S_{ij}(d)}{\lambda_{ij}}\Big)\frac{\partial \lambda_{ij}}{\partial z_i}.
\]
Substituting $\partial_{z_i}\lambda_{ij}=-\alpha s_{ij}\lambda_{ij}$ yields
\[
\frac{\partial \ell_{ij}}{\partial z_i}
=
\alpha s_{ij}\big(\widehat S_{ij}(d)-\lambda_{ij}(d)\big),
\qquad
\frac{\partial \ell_{ij}}{\partial z_j}
=
-\alpha s_{ij}\big(\widehat S_{ij}(d)-\lambda_{ij}(d)\big).
\]
Summing $\partial \ell_{ij}/\partial z_i$ over all neighbors $j\neq i$ gives the stated gradient for $\mathcal L_d(z)=\sum_{i<j}\ell_{ij}(z)$.

For the Hessian (still away from ties so $s_{ij}$ is locally constant), differentiate again:
\[
\frac{\partial^2 \ell_{ij}}{\partial z_i^2}
=
\frac{\partial}{\partial z_i}\Big(\alpha s_{ij}(\widehat S_{ij}-\lambda_{ij})\Big)
=
-\alpha s_{ij}\,\frac{\partial \lambda_{ij}}{\partial z_i}
=
\alpha^2\lambda_{ij},
\]
and for $i\neq j$,
\[
\frac{\partial^2 \ell_{ij}}{\partial z_i\partial z_j}
=
\frac{\partial}{\partial z_j}\Big(\alpha s_{ij}(\widehat S_{ij}-\lambda_{ij})\Big)
=
-\alpha s_{ij}\,\frac{\partial \lambda_{ij}}{\partial z_j}
=
-\alpha^2\lambda_{ij}.
\]
All other cross-partials are zero for this pair, so assembling contributions over pairs yields a weighted graph Laplacian structure.

\subsection{Laplace approximation and cold item uncertainty}
Let $\mathcal J_d(z)$ denote the MAP objective.
A Laplace approximation yields
\[
p\big(z(d)\mid y_{1:d}\big)\approx \mathcal N\!\big(\widehat z(d),\,\Sigma(d)\big),
\qquad
\Sigma(d)=\Big(\nabla^2 \mathcal J_d(\widehat z(d))\Big)^{-1}.
\]
Cold items have small exposure $e_{ij}(d)$, hence smaller $\lambda_{ij}(d)$, hence weaker curvature.
They retain larger posterior variance, which is the desired cold item behavior.

\paragraph{Derivation.}
Let $\mathcal J_d(z)$ be the negative log posterior objective (Section~5.3), so that
\[
p\big(z(d)\mid y_{1:d}\big) \propto \exp\big(-\mathcal J_d(z)\big).
\]
Assume $\mathcal J_d$ is twice differentiable in a neighborhood of the constrained MAP $\widehat z(d)$ (ignoring the measure-zero set of ties where $|z_i-z_j|$ is non-differentiable).
Using a second-order Taylor expansion around $\widehat z(d)$,
\[
\mathcal J_d(z)
=
\mathcal J_d\big(\widehat z(d)\big)
+\frac12\big(z-\widehat z(d)\big)^\top \nabla^2 \mathcal J_d\big(\widehat z(d)\big)\big(z-\widehat z(d)\big)
+\text{h.o.t.},
\]
where the linear term vanishes because $\widehat z(d)$ is a (constrained) minimizer.
Dropping the higher-order terms and exponentiating gives
\[
p\big(z(d)\mid y_{1:d}\big)
\approx
\exp\Big(-\mathcal J_d\big(\widehat z(d)\big)\Big)
\exp\!\Big(-\tfrac12\big(z-\widehat z(d)\big)^\top \nabla^2 \mathcal J_d\big(\widehat z(d)\big)\big(z-\widehat z(d)\big)\Big),
\]
which is proportional to a Gaussian density.
To see this, recall that the multivariate normal density $\mathcal N(\mu,\Sigma)$ can be written in canonical (information) form as
\[
\mathcal N(z\mid \mu,\Sigma)
\propto
\exp\!\Big(-\tfrac12(z-\mu)^\top \Sigma^{-1}(z-\mu)\Big),
\]
where the proportionality constant $(2\pi)^{-n/2}|\Sigma|^{-1/2}$ does not depend on $z$.
Comparing with the quadratic exponential term above, we identify
\[
\mu=\widehat z(d),
\qquad
\Sigma^{-1}=\nabla^2 \mathcal J_d\big(\widehat z(d)\big),
\]
so the Laplace approximation is Gaussian with mean $\widehat z(d)$ and precision matrix $\nabla^2 \mathcal J_d\big(\widehat z(d)\big)$.
Therefore the Laplace covariance is
\[
\Sigma(d)=\Big(\nabla^2 \mathcal J_d\big(\widehat z(d)\big)\Big)^{-1}.
\]
Finally, the data Hessian contribution has Laplacian weights $\alpha^2\lambda_{ij}(d)$ (Section~5.4), so when exposures $e_{ij}(d)$ are small for cold items, $\lambda_{ij}(d)$ is small and curvature is weak, implying larger posterior variances.

\section{Decision rule, ordering from a posterior}

\subsection{Point estimate ordering}
The posterior mode induces a permutation
\[
\widehat \pi(d)=\mathrm{argsort}(\widehat z(d)).
\]

\subsection{Conservative ordering using uncertainty}
To avoid aggressive promotion of uncertain cold items, define a lower confidence score
\[
s_i(d)=\widehat z_i(d)-\kappa\sqrt{\Sigma_{ii}(d)},
\qquad \kappa>0,
\]
and output a conservative order
\[
\pi_{\mathrm{LCB}}(d)=\mathrm{argsort}\big(s(d)\big).
\]
This mirrors conservative leaderboards in skill rating systems.

\section{Practical solver}

\subsection{Daily inference algorithm (see Appendix)}
\paragraph{Inputs.}
For each day $d$, assume access to the sessions on day $d$, the previous day approximation
$\big(\widehat z(d-1),\Sigma(d-1)\big)$, and hyperparameters $(\alpha,\tau^2,\rho_c,\tau_e,\kappa)$.

\paragraph{Outputs.}
Daily posterior approximation $\mathcal N\!\big(\widehat z(d),\Sigma(d)\big)$ and an ordering (either $\widehat\pi(d)$ or $\pi_{\mathrm{LCB}}(d)$).

The full day-by-day procedure is given in Algorithm~\ref{alg:daily_inference} in Appendix~\ref{app:daily_inference}.

\subsection{Computational notes}
The Hessian is a sum of a diagonal prior precision and a Laplacian structured matrix from the data.
With sparse counts, only observed edges contribute.
This structure supports fast iterative linear solves inside Newton steps.

\subsection{Hyperparameter selection via grid search and bandits}
The model introduces hyperparameters controlling drift, exposure discounting, and conservatism, for example
$(\tau^2,\alpha,\tau_e,\rho_c,\kappa)$, and possibly the day intercept rule for $\beta(d)$.
In practice these should be tuned against an offline proxy objective (e.g., next-day predictive log likelihood of counts,
agreement with held-out session edits, or stability/latency metrics for a deployed ranking).

\paragraph{Log-space grids.}
Many parameters act as scales and are naturally searched in log space.
For instance, choose grids such as
\[
\begin{aligned}
\log \tau^2 &\in \{\log 10^{-4},\log 10^{-3},\dots,\log 10^{0}\},\\
\log \alpha &\in \{\log 10^{-2},\log 10^{-1.5},\dots,\log 10^{1}\},\\
\log \tau_e &\in \{\log 10^{-2},\dots,\log 10^{2}\},
\end{aligned}
\]
while using linear grids for bounded parameters such as $\rho_c\in(0,1)$ (e.g., $\rho_c\in\{0.8,0.9,0.95,0.98\}$)
and nonnegative multipliers such as $\kappa$ (e.g., $\kappa\in\{0,0.5,1,2\}$).
When a full Cartesian grid is too large, use a coarse grid followed by a local refinement around the best region.

\paragraph{Rolling-window evaluation.}
To reflect nonstationarity, evaluate each candidate setting on rolling windows.
For example, for each day $d$ in a validation range, fit using data up to day $d$ and score on day $d+1$
(or score on a held-out subset of edges/pairs within day $d$).
Aggregate scores with an exponential decay so that recent performance dominates:
\[
\mathrm{Score}(\theta)
=
\sum_{d\in\mathcal D_{\mathrm{val}}} \gamma^{(d_{\max}-d)}\,\mathrm{score}_d(\theta),
\qquad 0<\gamma<1,
\]
where $\theta$ denotes a hyperparameter setting.

\paragraph{Bandit allocation over hyperparameter ``arms.''}
Instead of spending equal compute on all grid points, treat each candidate $\theta$ as an arm in a multi-armed bandit.
Each day (or each evaluation batch), allocate a fixed compute budget by selecting a subset of arms to evaluate on the newest data.
Let $r_d(\theta)$ be the reward (e.g., negative predictive loss, click-through proxy, or stability-adjusted utility).
Maintain exponentially-decayed statistics
\[
\bar r_d(\theta)=\eta\bar r_{d-1}(\theta)+(1-\eta)r_d(\theta),
\qquad
N_d(\theta)=\eta N_{d-1}(\theta)+(1-\eta),
\]
with $0<\eta<1$.
Selection can follow an explore--exploit rule such as decayed-UCB,
\[
\theta_d\in\arg\max_{\theta}\ \bar r_d(\theta)+c\sqrt{\frac{\log(1+d)}{\max\{N_d(\theta),\epsilon\}}}
\]
or an exponential-weights (Hedge) update,
\[
p_d(\theta)\propto p_{d-1}(\theta)\exp\big(\lambda\,r_d(\theta)\big),
\]
where $c,\epsilon,\lambda$ are tuning constants.
The exponential decay emphasizes current conditions and allows rapid adaptation when user behavior shifts.

\paragraph{Practical recipe.}
Start with a small log-grid over $(\tau^2,\alpha,\tau_e)$ and a short list for $(\rho_c,\kappa)$.
Run a few days of uniform exploration to initialize rewards, then switch to a decayed-UCB or exponential-weights bandit
to focus compute on the most promising regions while still occasionally probing alternatives.


\section{Discussion}
This paper framed the ordering problem as inference of a drifting hidden preference state.
Cold items were treated as a source of uncertainty rather than as deterministic stiffness.
The resulting posterior provides both a daily best estimate order and calibrated uncertainty that can be used for conservative deployment rules.

\appendix
\section{Daily inference algorithm}
\label{app:daily_inference}
\begin{algorithm}[tbp]
\centering
\caption{Daily MAP + Laplace update for dynamic latent order inference}
\label{alg:daily_inference}
\begin{enumerate}
\item \textbf{Build daily sufficient statistics.}
From sessions on day $d$, compute the symmetric consecutive update counts $\widehat S_{ij}(d)$.
Also compute per-item update counts $\widehat c_i(d)$, then update the decayed mass
$c_i(d)=\rho_c c_i(d-1)+(1-\rho_c)\widehat c_i(d)$ and days-since-activity $t_i(d)$.

\item \textbf{Compute exposures and Poisson intensities.}
Form per-item exposures $\mathrm{expo}_i(d)=c_i(d)/(c_i(d)+\tau_e)$ and pair exposures
$e_{ij}(d)=\mathrm{expo}_i(d)\mathrm{expo}_j(d)$.
Given a current iterate $z$, define $\delta_{ij}(d)=|z_i-z_j|$ and
$\lambda_{ij}(d)=e_{ij}(d)\exp\big(\beta(d)-\alpha\delta_{ij}(d)\big)$.
In practice, $\beta(d)$ may be set by matching total mass or treated as an optimizable intercept.

\item \textbf{Predict by prior propagation.}
Set
\[
Q(d)=\mathrm{diag}\big(\tau^2 t_1(d),\dots,\tau^2 t_n(d)\big),
\qquad
m^-(d)=\widehat z(d-1),
\]
and use $P^-(d)=\Sigma(d-1)+Q(d)$.

\item \textbf{Form the day-$d$ objective.}
Define the constrained MAP objective
\[
\mathcal J_d(z)=\mathcal L_d(z)+\frac12\big(z-m^-(d)\big)^\top\big(P^-(d)\big)^{-1}\big(z-m^-(d)\big),
\qquad \mathbf 1^\top z=0.
\]

\item \textbf{Solve the MAP problem.}
Initialize $z^{(0)}\leftarrow m^-(d)$, projected to $\mathbf 1^\top z=0$ if necessary.
For $k=0,1,2,\dots$ until convergence, compute
\[
\nabla\mathcal J_d(z)=\nabla\mathcal L_d(z)+\big(P^-(d)\big)^{-1}(z-m^-(d)),
\qquad
\nabla^2\mathcal J_d(z)=\nabla^2\mathcal L_d(z)+\big(P^-(d)\big)^{-1},
\]
then solve $\nabla^2\mathcal J_d(z)\,\Delta z=-\nabla\mathcal J_d(z)$, update $z\leftarrow z+\Delta z$, and re-center $z$.

\item \textbf{Laplace covariance update.}
Set $\Sigma(d)=\big(\nabla^2\mathcal J_d(\widehat z(d))\big)^{-1}$, or compute only selected entries by linear solves when $n$ is large.

\item \textbf{Output an ordering.}
Return either $\widehat\pi(d)=\mathrm{argsort}(\widehat z(d))$ or the conservative order
$\pi_{\mathrm{LCB}}(d)=\mathrm{argsort}(s(d))$, where $s_i(d)=\widehat z_i(d)-\kappa\sqrt{\Sigma_{ii}(d)}$.
\end{enumerate}
\end{algorithm}
